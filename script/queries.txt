

UPDATE ADDRESSES AS A ,ADDRESS_UNIQUE  AS AU
SET A.LATLONGID=AU.LATLONGID
WHERE AU.INSTITUTE = A.INSTITUTION  AND
      AU.VILLE     = A.VILLE  AND
      AU.PROVINCE  = A.PROVINCE AND
      AU.COUNTRY   = A.COUNTRY
	  
	  
	  
UPDATE CITED_PAPER_ADDRESS AS CPA ,ADDRESS_UNIQUE  AS AU
SET   CPA.LATLONGID=AU.LATLONGID
WHERE AU.INSTITUTE = CPA.INSTITUTION  AND
      AU.VILLE     = CPA.VILLE  AND
      AU.PROVINCE  = CPA.PROVINCE AND
      AU.COUNTRY   = CPA.COUNTRY
